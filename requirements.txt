#   python -c "import torch; print(torch.cuda.is_available())"
#

# ============================================================
# Installation Instructions
# ============================================================
#
# STEP 1: Install ROCm (for AMD GPU)
#   wget https://repo.radeon.com/amdgpu-install/latest/ubuntu/focal/amdgpu-install_*.deb
#   sudo apt install ./amdgpu-install_*.deb
#   sudo amdgpu-install --usecase=rocm
#
# STEP 2: Install PyTorch with ROCm
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/rocm6.0
#
# STEP 3: Install remaining requirements
#   pip install -r requirements.txt
#
# STEP 4: Verify installation
#   python -c "import torch; print(f'GPU: {torch.cuda.is_available()}')"
#   python -c "import dash; import h5py; print('All deps OK')"
#
# STEP 5: Test radar app
#   python main.py --simulate --duration 10
#
# ============================================================
# Alternative: CPU-Only Installation
# ============================================================
# If no GPU or ROCm issues:
#   pip install torch torchvision torchaudio
#   pip install -r requirements.txt
#
# System will work on CPU (slower but functional)
#
# ============================================================
# Troubleshooting
# ============================================================
#
# Issue: "CUDA not available"
# → Check: rocm-smi
# → Reinstall PyTorch with correct ROCm version
#
# Issue: "No module named 'adi'"
# → pip install pyadi-iio
# → Or run ./fix_libiio.sh
#
# Issue: "Import error: dash"
# → pip install dash plotly
#
